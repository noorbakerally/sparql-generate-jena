20  import org.apache.jena.query.Query;
21  import org.apache.jena.query.QueryVisitor;
22  import org.apache.jena.sparql.core.Prologue;

29  
30  /**
31   * Extends the ARQ Query Serializer with SPARQL Generate specificities.
32   * 
33   * @author maxime.lefrancois
34   */
35  public class SPARQLGenerateQuerySerializer implements com.github.thesmartenergy.sparql.generate.jena.query.SPARQLGenerateQueryVisitor {
36  
37      public final int BLOCK_INDENT = 2;
38      private final QueryVisitor decorated;
39      private final IndentedWriter out;
40      private final SPARQLGenerateFormatterElement fmtElement;
41      private final FmtExprSPARQL fmtExpr;
42      private final FormatterTemplate fmtTemplate;
43  
44      public SPARQLGenerateQuerySerializer(QueryVisitor serializer, OutputStream _out, SPARQLGenerateFormatterElement formatterElement, FmtExprSPARQL formatterExpr, FormatterTemplate formatterTemplate) {
45          this(serializer, new IndentedWriter(_out), formatterElement, formatterExpr, formatterTemplate);
46      }
47  
48      public SPARQLGenerateQuerySerializer(QueryVisitor serializer, IndentedWriter iwriter, SPARQLGenerateFormatterElement formatterElement, FmtExprSPARQL formatterExpr, FormatterTemplate formatterTemplate) {
49          decorated = serializer;
50          out = iwriter;
51          fmtTemplate = formatterTemplate;
52          fmtElement = formatterElement;
53          fmtExpr = formatterExpr;
54      }
55  
56      @Override
57      public void visitGenerateResultForm(com.github.thesmartenergy.sparql.generate.jena.query.SPARQLGenerateQuery query) {
58          out.print("GENERATE ");
59          if (query.hasGenerateURI()) {
60              out.print(" ");
61              out.print("<" + query.getGenerateURI() + ">");
62              out.newline();
63          } else if (query.hasGenerateTemplate()) {
64              out.newline();
65              out.incIndent(BLOCK_INDENT);
66              fmtElement.visit(query.getGenerateTemplate());
67              out.decIndent(BLOCK_INDENT);
68              out.newline();
69          }
70      }
71      
72      @Override
73      public void visitIteratorsAndSources(SPARQLGenerateQuery query) {
74          if(query.getIteratorsAndSources() == null) {
75              return;
76          }
77          for (ElementIteratorOrSource iteratorOrSource : query.getIteratorsAndSources()) {
78              if(iteratorOrSource instanceof ElementIterator) {
79                  fmtElement.visit((ElementIterator) iteratorOrSource);
80              } else if(iteratorOrSource instanceof ElementSource) {
81                  fmtElement.visit((ElementSource) iteratorOrSource);
82              }
83              out.newline();
84          }
85      }
86  
87      @Override
88      public void startVisit(Query query) {
89          decorated.startVisit(query);
90      }
91  
92      @Override
93      public void visitPrologue(Prologue prologue) {
94          decorated.visitPrologue(prologue);
95      }
96  
97      @Override
98      public void visitResultForm(Query query) {
99          decorated.visitResultForm(query);
100     }
101

120     }

125     }

129         decorated.visitQueryPattern(query);
130     }
131

135     }

138     public void visitHaving(Query query) {
139         decorated.visitHaving(query);
140     }
141 
142     @Override
143     public void visitOrderBy(Query query) {
144         decorated.visitOrderBy(query);
145     }
146

148     public void visitLimit(Query query) {
149         decorated.visitLimit(query);
150     }
151

153     public void visitOffset(Query query) {

156     }
157

159     public void visitValues(Query query) {
160         decorated.visitOffset(query);
161     }
162

164     public void finishVisit(Query query) {
165         decorated.finishVisit(query);
166     }

168 }

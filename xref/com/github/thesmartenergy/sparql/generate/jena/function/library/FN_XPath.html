18  import com.github.thesmartenergy.sparql.generate.jena.SPARQLGenerate;
19  import java.io.ByteArrayInputStream;
20  import java.io.InputStream;
21  import javax.xml.parsers.DocumentBuilder;
22  import javax.xml.parsers.DocumentBuilderFactory;
23  import javax.xml.xpath.XPath;
24  import javax.xml.xpath.XPathConstants;
25  import javax.xml.xpath.XPathFactory;
26  import org.apache.jena.graph.Node;
27  import org.apache.jena.sparql.expr.ExprEvalException;
28  import org.apache.jena.sparql.expr.NodeValue;
29  import org.apache.jena.sparql.expr.nodevalue.NodeValueString;
30  import org.apache.jena.sparql.function.FunctionBase2;
31  import org.apache.log4j.Logger;
32  import org.w3c.dom.Document;

34  /**
35   * A SPARQL Function that extracts a string from a XML document, according to a
36   * XPath expression. The Function URI is
37   * {@code <http://w3id.org/sparql-generate/fn/XPath>}.
38   * It takes two parameters as input:
39   * <ul>
40   * <li>a RDF Literal with datatype URI
41   * {@code <urn:iana:mime:application/json>}</li>
42   * <li>a RDF Literal with datatype {@code xsd:string}</li>
43   * </ul>
44   * and returns a RDF Literal with datatype URI
45   * {@code <urn:iana:mime:application/json>}.
46   *
47   * @author maxime.lefrancois
48   */
49  public class FN_XPath extends FunctionBase2 {
50      //TODO write multiple unit tests for this class.
51  
52      /**
53       * The logger.
54       */
55      private static final Logger LOG = Logger.getLogger(FN_XPath.class);
56  
57      /**
58       * The SPARQL function URI.
59       */
60      public static final String URI = SPARQLGenerate.FN + "XPath";
61  
62      /**
63       * The datatype URI of the first parameter and the return literals.
64       */
65      private static final String datatypeUri = "urn:iana:mime:application/xml";
66  
67      /**
68       * Returns the evaluation of XPath {@code xpath} over the XML
69       * document {@code xml}.
70       * @param xml the RDF Literal that represents a XML document
71       * @param xpath the xsd:string that represents the XPath
72       * @return -
73       */
74      @Override
75      public NodeValue exec(NodeValue xml, NodeValue xpath) {
76          if (xml.getDatatypeURI() == null
77                  && datatypeUri == null
78                  || xml.getDatatypeURI() != null
79                  && !xml.getDatatypeURI().equals(datatypeUri)
80                  && !xml.getDatatypeURI().equals("http://www.w3.org/2001/XMLSchema#string")) {
81              LOG.warn("The URI of NodeValue1 MUST be <" + datatypeUri + ">"
82                      + "or <http://www.w3.org/2001/XMLSchema#string>."
83                      + " Returning null.");
84          }
85  
86          DocumentBuilderFactory builderFactory
87                  = DocumentBuilderFactory.newInstance();
88          DocumentBuilder builder = null;
89          try {
90              builder = builderFactory.newDocumentBuilder();
91              InputStream in = new ByteArrayInputStream(
92                      xml.asNode().getLiteralLexicalForm().getBytes());
93              Document document = builder.parse(in);
94  
95              XPath xPath =  XPathFactory.newInstance().newXPath();
96              Node node = (Node) xPath.compile(xpath.getString())
97                      .evaluate(document, XPathConstants.NODE);
98  
99              return new NodeValueString((String) node.toString());
100         } catch (Exception e) {
101             throw new ExprEvalException("FunctionBase: no evaluation", e);
102         }
103     }
104 }

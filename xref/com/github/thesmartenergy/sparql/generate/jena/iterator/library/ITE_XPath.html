39  /**
40   * A SPARQL Iterator function that extracts a list of sub-XML elements of a
41   * XML root element, according to a XPath expression. The Iterator function URI is
42   * {@code <http://w3id.org/sparql-generate/ite/XPath>}.
43   * It takes two parameters as input:
44   * <ul>
45   * <li>a RDF Literal with datatype URI
46   * {@code <urn:iana:mime:application/xml>}</li>
47   * <li>a RDF Literal with datatype {@code xsd:string}</li>
48   * </ul>
49   * and returns a list of RDF Literal with datatype URI
50   * {@code <urn:iana:mime:application/xml>}.
51   *
52   * @author maxime.lefrancois
53   */
54  public class ITE_XPath extends IteratorFunctionBase2 {
55  
56      /**
57       * The logger.
58       */
59      private static final Logger LOG = Logger.getLogger(ITE_XPath.class);
60  
61      /**
62       * The SPARQL function URI.
63       */
64      public static final String URI = SPARQLGenerate.ITE + "XPath";
65  
66      /**
67       * The datatype URI of the first parameter and the return literals.
68       */
69      private static final String datatypeUri = "urn:iana:mime:application/xml";
70  
71      @Override
72      public List<NodeValue> exec(NodeValue xml, NodeValue v2) {
73          if (xml.getDatatypeURI() == null
74                  && datatypeUri == null
75                  || xml.getDatatypeURI() != null
76                  && !xml.getDatatypeURI().equals(datatypeUri)
77                  && !xml.getDatatypeURI().equals("http://www.w3.org/2001/XMLSchema#string")) {
78              LOG.warn("The URI of NodeValue1 MUST be"
79                      + " <" + datatypeUri + "> or"
80                      + " <http://www.w3.org/2001/XMLSchema#string>. Got <"
81                      + xml.getDatatypeURI() + ">. Returning null.");
82          }
83          DocumentBuilderFactory builderFactory
84                  = DocumentBuilderFactory.newInstance();
85          DocumentBuilder builder = null;
86          try {
87              builder = builderFactory.newDocumentBuilder();
88              Document document = builder
89                      .parse(new ByteArrayInputStream(
90                              xml.asNode().getLiteralLexicalForm().getBytes()));
91              
92              XPath xPath =  XPathFactory.newInstance().newXPath();
93              
94              NodeList nodeList = (NodeList) xPath
95                      .compile(v2.getString())
96                      .evaluate(document, XPathConstants.NODESET);
97              List<NodeValue> nodeValues = new ArrayList<>(nodeList.getLength());
98  
99              for (int i=0;i<nodeList.getLength();i++) {
100                 org.w3c.dom.Node xmlNode = nodeList.item(i);
101                 System.out.println("SEL --> "+xmlNode);
102                 RDFDatatype dt = TypeMapper.getInstance()
103                         .getSafeTypeByName(datatypeUri);
104                 
105                 Node node = NodeFactory.createLiteral(xmlNode.toString(), dt);
106                 NodeValue nodeValue = new NodeValueNode(node);
107                 nodeValues.add(nodeValue);
108             }
109             return nodeValues;
110         } catch (Exception e) {
111             throw new ExprEvalException("FunctionBase: no evaluation", e);
112         }
113     }
114 }

16  package com.github.thesmartenergy.sparql.generate.jena.engine;
17  
18  import com.github.thesmartenergy.sparql.generate.jena.engine.impl.BindingHashMapOverwrite;
19  import java.util.List;
20  import org.apache.jena.query.Dataset;
21  import org.apache.jena.rdf.model.Model;
22  import org.apache.jena.sparql.core.Var;
23  
24  /**
25   * The GENERATE Clause
26   * @author maxime.lefrancois
27   */
28  public interface GeneratePlan {
29  
30      /**
31       * Executes the GENERATE Clause. Uses the given {@code inputDataset}
32       * for the subsequent SPARQL GENERATE queries. Augments the given
33       * {@code initialModel} with the generated RDF triples. Use the given
34       * {@code variables} and {@code values} to generate triples. The behaviour
35       * if {@code initialModel} is one of the RDF graphs in {@code inputDataset}
36       * is not specified.
37       *
38       * @param inputDataset the Dataset to use for the SPARQL SELECT part of the
39       * query.
40       * @param initialModel the Model to augment with the generated RDF triples.
41       * @param variables the set of bound variables.
42       * @param values the set of bindings.
43       * @throws IllegalArgumentException if the {@code initialModel} is null.
44       */
45      void exec (
46          final Dataset inputDataset,
47          final Model initialModel,
48          final List<Var> variables,
49          final List<BindingHashMapOverwrite> values);
50  
51  }

32  /**
33   * The GENERATE {...} template.
34   *
35   * @author maxime.lefrancois
36   */
37  public class GenerateTemplatePlanImpl extends PlanBase implements GeneratePlan {
38  
39      /**
40       * the logger.
41       */
42      private static final Logger LOG
43              = Logger.getLogger(GenerateTemplatePlanImpl.class);
44  
45      /**
46       * the list of generate blocks.
47       */
48      private final List<GenerateTemplateElementPlan> templateElementPlans;
49  
50      /**
51       * Constructs a new Generate plan from the given list of generate block
52       * plans.
53       *
54       * @param templateElementPlans a list of plans
55       */
56      public GenerateTemplatePlanImpl(
57              final List<GenerateTemplateElementPlan> templateElementPlans) {
58          this.templateElementPlans = templateElementPlans;
59      }
60  
61      /**
62       * gets the list of plans for generate blocks.
63       *
64       * @return a list of plans
65       */
66      public List<GenerateTemplateElementPlan> getTemplateElementsPlans() {
67          return templateElementPlans;
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public final void exec(
75              final Dataset inputDataset,
76              final Model initialModel,
77              final List<Var> variables,
78              final List<BindingHashMapOverwrite> values) {
79          LOG.debug("exec");
80          //TODO write unit tests for bNodes
81          for (BindingHashMapOverwrite binding : values) {
82              // create new blank node map
83              Map<Node, Node> bNodeMap = new HashMap<>();
84              for (Var v : binding.varsList()) {
85                  Node n = binding.get(v);
86                  if (n.isBlank()) {
87                      Node bn = NodeFactory.createBlankNode();
88                      bNodeMap.put(n, bn);
89                  }
90              }
91              for (GenerateTemplateElementPlan el : templateElementPlans) {
92                  if (el instanceof GenerateTriplesPlanImpl) {
93                      GenerateTriplesPlanImpl subPlanTriples
94                              = (GenerateTriplesPlanImpl) el;
95                      subPlanTriples.exec(
96                              inputDataset, initialModel,
97                              binding, bNodeMap);
98                  } else if (el instanceof RootPlanImpl) {
99                      RootPlanImpl rootPlan = (RootPlanImpl) el;
100                     QuerySolutionMap b = new QuerySolutionMap();
101                     for (Var v : binding.varsList()) {
102                         Node n = binding.get(v);
103                         if (bNodeMap.containsKey(n)) {
104                             b.add(v.getVarName(), initialModel.asRDFNode(bNodeMap.get(n)));
105                         } else {
106                             b.add(v.getVarName(), initialModel.asRDFNode(n));
107                         }
108                     }
109                     rootPlan.exec(inputDataset, b, initialModel);
110                 } else {
111                     throw new SPARQLGenerateException("should not reach this point");
112                 }
113             }
114         }
115     }

36  /**
37   * Executes the SPARQL SELECT query.
38   *
39   * @author maxime.lefrancois
40   */
41  public class SelectPlanImpl extends PlanBase implements SelectPlan {
42  
43      private static final Logger LOG = Logger.getLogger(SelectPlanImpl.class);
44  
45      /**
46       * The query.
47       */
48      private final Query select;
49  
50      /**
51       * Constructor.
52       *
53       * @param query the SPARQL SELECT query.
54       */
55      public SelectPlanImpl(final Query query) {
56          if (!query.isSelectType()) {
57              throw new IllegalArgumentException("Should be select query.");
58          }
59          this.select = query;
60      }
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      final public void exec(
67              final Dataset inputDataset,
68              final List<Var> variables,
69              final List<BindingHashMapOverwrite> values) {
70          LOG.debug("exec");
71  
72          Query q = select.cloneQuery();
73          
74          // add data block in the where clause.
75          ElementData data = new ElementData();
76          
77          for (Var v : variables) {
78              data.add(v);
79          }
80          for (BindingHashMapOverwrite b : values) {
81              data.add(b);
82          }
83          ElementGroup old = (ElementGroup) q.getQueryPattern();
84          
85          ElementGroup temp = new ElementGroup();
86          temp.addElement(data);
87          for(Element olde : old.getElements()) {
88              temp.addElement(olde);
89          }
90          q.setQueryPattern(temp);
91  //        LOG.debug("Resulting SELECT query:\n" + q);
92  
93          try {
94              QueryExecution exec = QueryExecutionFactory.create(q, inputDataset);
95              ResultSet results = exec.execSelect(); // delete this
96              LOG.debug(ResultSetFormatter.asText(results)); // delete this
97              
98              exec = QueryExecutionFactory.create(q, inputDataset);
99              results = exec.execSelect();
100             List<Var> newVariables = new ArrayList<>();
101             List<Binding> newValues = new ArrayList<>();
102             for (String var : results.getResultVars()) {
103                 newVariables.add(Var.alloc(var));
104             }
105             while (results.hasNext()) {
106                 newValues.add(new BindingHashMapOverwrite(results.next()));
107             }
108             
109             variables.clear();
110             values.clear();
111             for(Var newVar : newVariables) {
112                 variables.add(newVar);
113             }
114             for(Binding newBinding : newValues) {
115                 values.add((BindingHashMapOverwrite) newBinding);
116             }
117         } catch (Exception ex) {
118             throw new SPARQLGenerateException("Error while executing"
119                     + " SELECT Query", ex);
120         }
121     }
122 
123     /**
124      * merge values in qvalues.
125      *
126      * @param qvariables
127      * @param qvalues
128      * @param variables
129      * @param values
130      */
131     private void mergeValues(
132             final List<Var> qvariables,
133             final List<Binding> qvalues,
134             final List<Var> variables,
135             final List<BindingHashMapOverwrite> values) {
136 
137         // check that no variable is duplicate in variables and qvariables
138         // brute force, ok if not many variables.
139         checkNotContainsSome(qvariables, variables);
140         checkNotContainsSome(variables, qvariables);
141 
142         // merge variables
143         qvariables.addAll(variables);
144 
145         // deal with empty cases
146         if (values.isEmpty() && qvalues.isEmpty()) {
147             return;
148         }
149         // ensure not empty
150         ensureNotEmpty(qvariables, qvalues);
151         ensureNotEmpty(variables, values);
152 
153         // perform the merge
154         int j = qvalues.size();
155         for (int i = 0; i < j; i++) {
156             Binding qb = qvalues.remove(0);
157             for (Binding b : values) {
158                 BindingHashMap newqb = new BindingHashMap();
159                 newqb.addAll(qb);
160                 newqb.addAll(b);
161                 qvalues.add(newqb);
162             }
163         }
164     }
165 
166     private void checkNotContainsSome(List<Var> vars1, List<Var> vars2) {
167         for (Var v : vars1) {
168             if (vars2.contains(v)) {
169                 throw new SPARQLGenerateException("Variable " + v + " has"
170                         + " already been bound !");
171             }
172         }
173     }

16  package org.w3id.sparql.generate.function.library;
17  
18  import java.io.ByteArrayInputStream;
19  import org.w3id.sparql.generate.selector.library.SEL_JSONPath;
20  import javax.xml.parsers.DocumentBuilder;
21  import javax.xml.parsers.DocumentBuilderFactory;
22  import javax.xml.xpath.XPath;
23  import javax.xml.xpath.XPathConstants;
24  import javax.xml.xpath.XPathFactory;
25  import org.apache.jena.graph.Node;
26  import org.apache.jena.sparql.expr.ExprEvalException;
27  import org.apache.jena.sparql.expr.NodeValue;
28  import org.apache.jena.sparql.expr.nodevalue.NodeValueString;
29  import org.apache.jena.sparql.function.FunctionBase2;
30  import org.apache.log4j.Logger;
31  import org.w3c.dom.Document;
32  
33  /**
34   *
35   * @author maxime.lefrancois
36   */
37  public class FN_XPath extends FunctionBase2 {
38  
39      private static final String uri = "urn:iana:mime:application/json";
40  
41      @Override
42      public NodeValue exec(NodeValue v1, NodeValue v2) {
43          if (v1.getDatatypeURI() == null ? uri != null : !v1.getDatatypeURI().equals(uri)) {
44              Logger.getLogger(SEL_JSONPath.class).warn("The URI of NodeValue1 MUST be <" + uri + ">. Returning null.");
45          }
46  
47          DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
48          DocumentBuilder builder = null;
49          try {
50              builder = builderFactory.newDocumentBuilder();
51              Document document = builder.parse(new ByteArrayInputStream(v1.asNode().getLiteralLexicalForm().getBytes()));
52              
53              XPath xPath =  XPathFactory.newInstance().newXPath();
54              Node node = (Node) xPath.compile(v2.getString()).evaluate(document, XPathConstants.NODE);
55  
56              return new NodeValueString((String) node.toString());
57          } catch (Exception e) {
58              throw new ExprEvalException("FunctionBase: no evaluation", e);
59          }
60      }
61  }

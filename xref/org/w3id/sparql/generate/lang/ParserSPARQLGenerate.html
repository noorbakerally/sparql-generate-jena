16  package org.w3id.sparql.generate.lang;
17  
18  import java.io.Reader ;
19  import java.io.StringReader ;
20  
21  import org.apache.jena.atlas.logging.Log ;
22  import org.apache.jena.query.Query;
23  import org.apache.jena.query.QueryException;
24  import org.apache.jena.query.QueryParseException;
25  import org.apache.jena.shared.JenaException;
26  import org.apache.jena.sparql.lang.SPARQLParser;
27  import org.w3id.sparql.generate.query.SPARQLGenerateQuery;
28  
29  /**
30   *
31   * @author maxime.lefrancois
32   */
33  public class ParserSPARQLGenerate extends SPARQLParser
34  {
35      private interface Action { void exec(SPARQLGenerateParser parser) throws Exception ; }
36      
37      @Override
38      protected Query parse$(final Query query, String queryString)
39      {
40          
41          SPARQLGenerateQuery q = new SPARQLGenerateQuery();
42          q.setSyntax(query.getSyntax());
43          q.setResolver(query.getResolver());
44  
45          Action action = new Action() {
46              @Override
47              public void exec(SPARQLGenerateParser parser) throws Exception
48              {
49                  parser.GenerateUnit() ;
50              }
51          } ;
52  
53          perform(q, queryString, action) ;
54          validateParsedQuery(q) ;
55          return q ;
56      }
57      
58      // All throwable handling.
59      private static void perform(Query query, String string, Action action)
60      {
61          Reader in = new StringReader(string) ;
62          SPARQLGenerateParser parser = new SPARQLGenerateParser(in) ;
63  
64          try {
65              query.setStrict(true) ;
66              parser.setQuery(query) ;
67              action.exec(parser) ;
68          }
69          catch (org.w3id.sparql.generate.lang.ParseException ex)
70          { 
71              throw new QueryParseException(ex.getMessage(),
72                                            ex.currentToken.beginLine,
73                                            ex.currentToken.beginColumn
74                                            ) ; }
75          catch (org.w3id.sparql.generate.lang.TokenMgrError tErr)
76          {
77              // Last valid token : not the same as token error message - but this should not happen
78              int col = parser.token.endColumn ;
79              int line = parser.token.endLine ;
80              throw new QueryParseException(tErr.getMessage(), line, col) ; }
81          
82          catch (QueryException ex) { throw ex ; }
83          catch (JenaException ex)  { throw new QueryException(ex.getMessage(), ex) ; }
84          catch (Error err)
85          {
86              // The token stream can throw errors.
87              throw new QueryParseException(err.getMessage(), err, -1, -1) ;
88          }
89          catch (Throwable th)
90          {
91              Log.warn(ParserSPARQLGenerate.class, "Unexpected throwable: ",th) ;
92              throw new QueryException(th.getMessage(), th) ;
93          }
94      }
95  }

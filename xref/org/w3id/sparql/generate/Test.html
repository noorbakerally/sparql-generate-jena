40  import org.w3id.sparql.generate.engine.GenerationPlanFactory;

44  /**
45   *
46   * @author maxime.lefrancois
47   */
48  public class Test {
49  
50      public static void main(String[] args) throws URISyntaxException, IOException {
51          SPARQLGenerate.init();
52  
53          URL examplesPath = Test.class.getResource("/example");
54          File file = new File(examplesPath.toURI());
55  
56          for (File exampleDir : file.listFiles()) {
57              System.out.println(exampleDir.toURI());
58  
59              // read location-mapping
60              URI confUri = exampleDir.toURI().resolve("configuration.ttl");
61              Model conf = RDFDataMgr.loadModel(confUri.toString());
62  
63              // initialize file manager
64              FileManager fileManager = FileManager.makeGlobal();
65              Locator loc = new LocatorFile(exampleDir.toURI().getPath());
66              LocationMapper mapper = new LocationMapper(conf);
67              fileManager.addLocator(loc);
68              fileManager.setLocationMapper(mapper);
69  
70              // parse query 
71              System.out.println("Processing example \"" + exampleDir.getName() + "\"");
72              String query = IOUtils.toString(fileManager.open("query.rqg"), "UTF-8");
73              SPARQLGenerateQuery q = (SPARQLGenerateQuery) QueryFactory.create(query, SPARQLGenerate.syntaxSPARQLGenerate);
74  
75              // serialize query 
76              URI queryOutputUri = exampleDir.toURI().resolve("query_serialized.rqg");
77              File queryOutputFile = new File(queryOutputUri);
78              OutputStream queryOutputStream = new FileOutputStream(queryOutputFile);
79              queryOutputStream.write(q.toString().getBytes());
80              queryOutputStream.close();
81  
82              // create generation plan
83              GenerationPlan plan = GenerationPlanFactory.create(q);
84              Model output = ModelFactory.createDefaultModel();
85                      GenerationQuerySolution initialBinding = new GenerationQuerySolution();
86  
87              // read inputs
88              NodeIterator nodes = conf.listObjectsOfProperty(conf.getProperty("http://w3id.org/sparql-generate/example#input"));
89              while (nodes.hasNext()) {
90                  Resource r = nodes.next().asResource();
91                  try {
92                      String varName = conf.listObjectsOfProperty(r, conf.getProperty("http://w3id.org/sparql-generate/example#var")).next().asLiteral().getLexicalForm();
93                      Var var = Var.alloc(varName);
94                      Resource contentType = conf.listObjectsOfProperty(r, conf.getProperty("http://w3id.org/sparql-generate/example#contentType")).next().asResource();
95                      String encoding = conf.listObjectsOfProperty(r, conf.getProperty("http://w3id.org/sparql-generate/example#encoding")).next().asLiteral().getLexicalForm();
96                      Resource location = conf.listObjectsOfProperty(r, conf.getProperty("http://w3id.org/sparql-generate/example#location")).next().asResource();
97  
98                      String inputString = IOUtils.toString(fileManager.open(location.getURI()), encoding);
99  
100                     Literal input = output.createTypedLiteral(inputString, contentType.getURI());
101                     initialBinding.put(varName, input);
102 
103                 } catch (Exception e) {
104                     Logger.getLogger(Test.class).info("unable to execute plan for input " + r);
105                 }
106             }
107 
108             // execute plan
109             plan.exec(initialBinding, output);
110 
111             // write output
112             URI outputUri = exampleDir.toURI().resolve("output.ttl");
113             File outputFile = new File(outputUri);
114             OutputStream outputStream = new FileOutputStream(outputFile);
115             output.write(outputStream, "TTL");
116             outputStream.close( );
117 
118             URI expectedOutputUri = exampleDir.toURI().resolve("expected_output.ttl");
119             Model expectedOutput = RDFDataMgr.loadModel(expectedOutputUri.toString());
120             expectedOutput.write(System.out, "TTL");
121             System.out.println("is equal: " + (output.containsAll(expectedOutput) && expectedOutput.containsAll(output)));
122         }
123     }
124 }

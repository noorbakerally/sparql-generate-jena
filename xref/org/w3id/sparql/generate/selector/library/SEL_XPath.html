23  import javax.xml.parsers.DocumentBuilderFactory;

26  import javax.xml.xpath.XPathFactory;

28  import org.apache.jena.graph.Node;
29  import org.apache.jena.sparql.expr.ExprEvalException;
30  import org.apache.jena.sparql.expr.NodeValue;
31  import org.apache.jena.sparql.expr.nodevalue.NodeValueNode;
32  import org.apache.log4j.Logger;
33  import org.w3c.dom.Document;
34  import org.w3c.dom.NodeList;
35  
36  /**
37   *
38   * @author maxime.lefrancois
39   */
40  public class SEL_XPath extends SelectorBase2 {
41  
42      private static final String iri = "urn:iana:mime:application/xml";
43  
44      public SEL_XPath() {
45          super(iri);
46      }
47  
48      @Override
49      public List<NodeValue> exec(NodeValue v1, NodeValue v2) {
50          if (v1.getDatatypeURI() == null ? iri != null : !v1.getDatatypeURI().equals(iri)) {
51              Logger.getLogger(SEL_XPath.class).warn("The URI of NodeValue1 MUST be <" + iri + ">. Returning null.");
52          }
53  
54          DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
55          DocumentBuilder builder = null;
56          try {
57              builder = builderFactory.newDocumentBuilder();
58              Document document = builder.parse(new ByteArrayInputStream(v1.asNode().getLiteralLexicalForm().getBytes()));
59              
60              XPath xPath =  XPathFactory.newInstance().newXPath();
61              
62              NodeList nodeList = (NodeList) xPath.compile(v2.getString()).evaluate(document, XPathConstants.NODESET);
63              List<NodeValue> nodeValues = new ArrayList<>(nodeList.getLength());
64  
65              for (int i=0;i<nodeList.getLength();i++) {
66                  org.w3c.dom.Node xmlNode = nodeList.item(i);
67                  System.out.println("SEL --> "+xmlNode);
68                  RDFDatatype dt = org.apache.jena.datatypes.TypeMapper.getInstance().getSafeTypeByName(iri);
69                  
70                  Node node = org.apache.jena.graph.NodeFactory.createLiteral(xmlNode.toString(), dt);
71                  NodeValue nodeValue = new NodeValueNode(node);
72                  nodeValues.add(nodeValue);
73              }
74              return nodeValues;
75          } catch (Exception e) {
76              throw new ExprEvalException("FunctionBase: no evaluation", e);
77          }
78      }
79  }
